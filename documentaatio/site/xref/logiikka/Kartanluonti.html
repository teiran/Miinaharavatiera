6   package logiikka;
7   
8   import logiikka.Ruutu;
9   
10  /**
11   * Luo Ruutu[][] kartan jossa on kaikki miinat ja luvut siitä kuinka monta miinaa on 
12   * naapurissa
13   *
14   * @author tiera
15   */
16  public class Kartanluonti {
17  
18      private static int korkeus;
19      private static int leveys;
20      private static int miinojenMaara;
21      private static Ruutu[][] kartta;
22      /**
23       * 
24       * @param korkeus kentän korkeus
25       * @param leveys kentän leveys
26       * @param miinojenMaara Miinojen määrä
27       * @return 
28       */
29      public static Ruutu[][] Kartanluonti(int korkeus, int leveys, int miinojenMaara) {
30          Kartanluonti.korkeus = korkeus;
31          Kartanluonti.leveys = leveys;
32          Kartanluonti.miinojenMaara = miinojenMaara;
33          kartta = new Ruutu[korkeus][leveys];
34          luokartta();
35          arvomiinat();
36          laskemiinat();
37          return kartta;
38  
39      }
40      
41      public static int miinojenmaara(){
42          return miinojenMaara;
43      }
44      
45  
46      private static void luokartta() {
47          for (int i = 0; i < korkeus; i++) {
48              for (int j = 0; j < leveys; j++) {
49                  kartta[i][j] = new Ruutu();
50              }
51          }
52      }
53  
54      private static void arvomiinat() {
55          int x = (int) (korkeus * Math.random());
56          int y = (int) (leveys * Math.random());
57          boolean t = (0 != miinojenMaara);
58          while (t) {
59              if (!kartta[x][y].isOnkoMiinaa()) {
60                  kartta[x][y].asetamiina();
61                  miinojenMaara--;
62              }
63              x = (int) (korkeus * Math.random());
64              y = (int) (leveys * Math.random());
65              t = (0 != miinojenMaara);
66          }
67      }
68  
69      private static void laskemiinat() {
70          for (int i = 0; i < korkeus; i++) {
71              for (int j = 0; j < leveys; j++) {
72                  if (!kartta[i][j].isOnkoMiinaa()) {
73                      laskemiinat(i, j);
74                  }
75              }
76          }
77      }
78  
79      private static void laskemiinat(int x, int y) {
80          int summa = 0;
81          for (int rajax = Math.max(x - 1, 0); rajax < Math.min(x + 2, korkeus); rajax++) {
82              for (int rajay = Math.max(y - 1, 0); rajay < Math.min(y + 2, leveys); rajay++) {
83                  if (kartta[rajax][rajay].isOnkoMiinaa()) {
84                      summa++;
85                  }
86              }
87  
88          }
89          kartta[x][y].laskemiinat(summa);
90      }
91  
92  }
