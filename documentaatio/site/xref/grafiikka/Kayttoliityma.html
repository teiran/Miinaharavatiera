6   package grafiikka;
7   
8   import java.awt.Container;
9   import java.awt.Dimension;
10  import java.awt.GridLayout;
11  import java.util.Scanner;
12  import javax.swing.JFrame;
13  import javax.swing.WindowConstants;
14  import logiikka.RuudukkoAvatutuSuljetut;
15  
16  /**
17   *
18   * @author tiera
19   */
20  public class Kayttoliityma implements Runnable{
21      
22      private JFrame frame;
23      private RuudukkoAvatutuSuljetut logiikka;
24      private RuutuButton[][] ruudut;
25      private int korkeus;
26      private int leveys;
27  
28      public Kayttoliityma() {
29          Scanner lukija = new Scanner(System.in);
30          System.out.println("Ruudukon leveys ");
31          korkeus = Integer.parseInt(lukija.nextLine());
32          System.out.println("Ruudukon korkeus ");
33          leveys = Integer.parseInt(lukija.nextLine());
34          System.out.println("Miinojen maara ");
35          int m = Integer.parseInt(lukija.nextLine());
36          logiikka = new RuudukkoAvatutuSuljetut(korkeus, leveys, m);
37          
38      }
39      
40      
41      
42      @Override
43      public void run() {
44          frame = new JFrame("Sudoku");
45          frame.setPreferredSize(new Dimension(1000, 1000));
46  
47          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
48  
49          luokomponetti(frame.getContentPane());
50  
51          frame.pack();
52          frame.setVisible(true);
53          
54          
55      }
56      
57      public void peli(){
58          for (int i = 0; i < korkeus; i++) {
59              for (int j = 0; j < leveys; j++) {
60                  if (ruudut[i][j].getPainettu() == 1) {
61                      logiikka.avaaruutu(i, j);
62                      ruudut[i][j].setPainettu(0);
63                  } else if (ruudut[i][j].getPainettu() == 2) {
64                      logiikka.lippu(i, j);
65                      ruudut[i][j].setPainettu(0);
66                  }
67              }
68          }
69          for (int i = 0; i < korkeus; i++) {
70              for (int j = 0; j < leveys; j++) {
71                  ruudut[i][j].setArvo(logiikka.kartta()[i][j]);
72              }
73          }
74      }
75      
76      private void luokomponetti(Container container){
77          frame.setLayout(new GridLayout(10, 10));
78          ruudut = new RuutuButton[korkeus][leveys];
79          for (int i = 0; i < korkeus; i++) {
80              for (int j = 0; j < leveys; j++) {
81                  ruudut[i][j] = new RuutuButton(logiikka.kartta()[i][j],this);
82                  Ruudunkuuntelija k = new Ruudunkuuntelija(ruudut[i][j]);
83                  ruudut[i][j].addMouseListener(k);
84                  container.add(ruudut[i][j]);
85              }
86          }
87      }
88      
89  }
