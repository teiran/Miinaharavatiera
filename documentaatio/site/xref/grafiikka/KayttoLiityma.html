6   package grafiikka;
7   
8   import java.awt.Container;
9   import java.awt.Dimension;
10  import java.awt.GridLayout;
11  import javax.swing.JFrame;
12  import javax.swing.WindowConstants;
13  import logiikka.Ruutu;
14  import logiikka.PelinLogiikka;
15  import static logiikka.Kartanluonti.kartanLuonti;
16  
17  /**
18   * pyörittää graafista käyttöliitymää, logiikkaan ja valikon.
19   *
20   * @author tiera
21   */
22  public class KayttoLiityma implements Runnable {
23  
24      private JFrame frame;
25      private int korkeus;
26      private int leveys;
27      private int miinojenmaara;
28      private final ValikkoPaneli valikko;
29      private final PeliPanelit peli;
30      private PelinLogiikka logiikka;
31  
32      /**
33       * Käyttöliitymänkonstuktori joka alustaa käyttöliitymän ja luo valikko ja peli panelit.
34       */
35      public KayttoLiityma() {
36          korkeus = 0;
37          leveys = 0;
38          miinojenmaara = 0;
39          peli = new PeliPanelit(this);
40          valikko = new ValikkoPaneli(this);
41  
42      }
43      /**
44       * Palaa logiikalta pelipanelille string muodossa arvot.
45       * 
46       * @param x logiikan ja grafiikan taulukon korkeus suuntainen koordinaatti
47       * @param y logiikan ja grafiikan taulukon leveys suuntainen koordinaatti
48       * @return palauttaa logiikalta saamansa arvon Stringmuodossa
49       */
50      public String getLogiikka(int x, int y) {
51          return logiikka.kartta()[x][y].toString();
52      }
53  
54      /**
55       * käyttöliitymän aloittava looppi joka alustaa alkuvalikon.
56       * 
57       */
58      @Override
59      public void run() {
60          frame = new JFrame("Miinaharava");
61          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
62          luoPelkkaValikko(frame.getContentPane());
63          frame.pack();
64          frame.setVisible(true);
65      }
66      /**
67       * päivittää valikon valintojen mukaisesksi kutsutaan uutta peliä luotaessa.
68       * 
69       */
70      public void paivitaGrafiikkaa() {
71          valikko.muutaSyoteet(korkeus, leveys, miinojenmaara);
72          frame.setVisible(false);
73          frame.removeAll();
74          frame = new JFrame("Miinaharava");
75          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
76          if (korkeus == 0 || leveys == 0) {
77              luoPelkkaValikko(frame.getContentPane());
78          } else {
79              luoValikkoJaPeli(frame.getContentPane());
80          }
81          frame.pack();
82          frame.setVisible(true);
83  
84      }
85      /**
86       * luo pelkällävalikolla varusetun version grafiikasta.
87       * 
88       * @param container sinne laitetaan panelit
89       */
90      private void luoPelkkaValikko(Container container) {
91          frame.setPreferredSize(new Dimension(500, 100));
92          valikko.paivita();
93          container.add(valikko.getPanel());
94      }
95      /**
96       * versio jossa on valikko ja varsinainen peli.
97       * 
98       * @param container paikka minne pannaa panelit
99       */
100     private void luoValikkoJaPeli(Container container) {
101         frame.setPreferredSize(new Dimension(50 * leveys + 50, 50 * korkeus));
102         container.setLayout(new GridLayout(korkeus + 1, 1));
103         valikko.paivita();
104         container.add(valikko.getPanel());
105         peli.paivita(korkeus, leveys);
106         Ruutu[][] f = kartanLuonti(korkeus, leveys, miinojenmaara);
107         logiikka = new PelinLogiikka(korkeus, leveys, f);
108         for (int i = 0; i < korkeus; i++) {
109             container.add(peli.luoRuudut(i));
110         }
111     }
112     
113     /**
114      * muokkaa korkeutta (kutsutaan valikkopanelista uutta kartta luodessa).
115      * 
116      * @param korkeus uusi pelikartan korkeus
117      */
118     public void setKorkeus(int korkeus) {
119         this.korkeus = korkeus;
120     }
121     /**
122      * muokkaa miinojenmäärää (kutsutaan valikkopanelista uutta kartta luodessa).
123      * 
124      * @param miinojenmaara uudenpelikartan miinojenmäärä
125      */
126     public void setMiinojenmaara(int miinojenmaara) {
127         this.miinojenmaara = miinojenmaara;
128     }
129     /**
130      * muokkaa leveyttä (kutsutaan valikkopanelista uutta kartta luodessa).
131      * 
132      * @param leveys uudenpelikartan leveys
133      */
134     public void setLeveys(int leveys) {
135         this.leveys = leveys;
136     }
137      /**
138       * linkki pelin "grafiikan" eli panelin ja logiikan välillä, kun painetaan nappulaa jollain kolmesta nappulasta tämä vie veistin grafiikalta logiikalle.
139       * 
140       * @param x logiikan ja grafiikan taulukon korkeus suuntainen koordinaatti
141       * @param y logiikan ja grafiikan taulukon leveys suuntainen koordinaatti
142       * @param oikeaVaiVasen numerosille että mitä nappulaa on painettu
143       */
144     public void paivitaAvaus(int x, int y, int oikeaVaiVasen) {
145         System.out.println(logiikka.muutaRuutua(x, y, oikeaVaiVasen));
146         paivitaRuudut();
147     }
148     
149     /**
150      * päivittää ruudut (grafiikan) muutetun logiikan mukaiseksi.
151      * 
152      */
153     private void paivitaRuudut() {
154         for (int i = 0; i < korkeus; i++) {
155             for (int j = 0; j < leveys; j++) {
156                 peli.paivitaRuudut(i, j, logiikka.kartta()[i][j].toString());
157             }
158         }
159         
160     }
161 }
